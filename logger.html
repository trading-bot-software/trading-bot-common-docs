"""
Used to log messages to stdout, file, redis and telegram.
"""

from datetime import datetime
from typing import Dict
from os import makedirs
from telegram import Bot
from traceback import format_exc
from asyncio import get_event_loop
from trading_bot_common.model.config import ConfigNames
from trading_bot_common.model.async_messages import RedisChannels
from trading_bot_common.redis import RedisConnection


class Logger:
    """
    Logs a messages to stdout, file, redis and telegram.
    Reads from config telegramKey, telegramChatId, outputFolder.
    - OUTPUT_FOLDER is required.
    - TELEGRAM_KEY, TELEGRAM_CHAT_ID are optional.
    """

    CONTROL = "CONTROL"
    WARN = "WARN"
    INFO = "INFO"
    ERROR = "ERROR"

    def __init__(self, config: Dict, redis_connection: RedisConnection = None):
        self.redis_connection = redis_connection
        self.telegram_bot: Bot = None
        self.chat_id: str = None
        self.config = config
        if (
            ConfigNames.TELEGRAM_KEY in config
            and ConfigNames.TELEGRAM_CHAT_ID in config
        ):
            self.chat_id = config[ConfigNames.TELEGRAM_CHAT_ID]
            self.telegram_bot: Bot = Bot(token=config[ConfigNames.TELEGRAM_KEY])

        self._validate_logging_folder()

    # region getter/setter

    def _validate_logging_folder(self):
        loggin_folder = "output/{}".format(self.config[ConfigNames.OUTPUT_FOLDER])
        makedirs(loggin_folder, exist_ok=True)

    def _get_logging_folder(self):
        return "output/{}".format(self.config[ConfigNames.OUTPUT_FOLDER])

    def read_lines_from_file(self, lastLines=-1):
        """
        Reads the last lines from the log file.
        """
        with open("{}/log.txt".format(self._get_logging_folder()), "r+") as f:
            data = f.read().splitlines(True)
        if lastLines != -1 and len(data) > lastLines:
            return "".join(data[len(data) - lastLines :])
        return "".join(data)

    # endregion

    # region print to

    def _print_to_telegram(self, msg: str):
        """
        Prints a message to telegram.
        """
        if self.telegram_bot is None:
            return
        msg = (
            msg[-4096:] if len(msg) > 4096 else msg
        )  # max telegram message length is 4096
        try:
            self.telegram_bot.send_message(chat_id=self.chat_id, text=msg)
        except:
            print("Cannot send message to telegram.")

    def _print_to_redis(self, msg: str):
        """
        Prints a message to redis.
        """
        if self.redis_connection == None:
            return

        get_event_loop().create_task(
            self.redis_connection.publish(RedisChannels.LOGS, msg)
        )

    def _print_to_file_and_stdout(self, msg: str):
        """
        Prints a message to file.
        """
        with open("{}/log.txt".format(self._get_logging_folder()), "a+") as f:
            f.write(msg + "\n")
        print(msg)

    # endregion

    # region print methods

    def _print_message(self, tag, *message_arr: object, send_telegram: bool = False):
        """
        Prints a message to stdout, file, telegram and redis.
        """

        info_message = ""
        for msg in message_arr:
            info_message += str(msg) + " "
        self._print_to_file_and_stdout(
            "{}: {}: {}".format(
                tag, datetime.utcnow().strftime("%m/%d/%Y-%H:%M:%S"), str(info_message)
            )
        )
        if send_telegram:
            self._print_to_telegram(info_message)
        self._print_to_redis(info_message)

    def print_error(self, *error_message_arr: object, send_telegram: bool = False):
        """
        Prints an error message.
        """
        self._print_message(
            Logger.ERROR, *error_message_arr, send_telegram=send_telegram
        )
        self._print_message(Logger.ERROR, format_exc(), send_telegram=send_telegram)

    def print_control(self, *info_message_arr: object, send_telegram: bool = False):
        """
        Prints a control message.
        """
        self._print_message(
            Logger.CONTROL, *info_message_arr, send_telegram=send_telegram
        )

    def print_info(self, *info_message_arr: object, send_telegram: bool = False):
        """
        Prints an info message.
        """
        self._print_message(Logger.INFO, *info_message_arr, send_telegram=send_telegram)

    def print_warn(self, *info_message_arr: object, send_telegram: bool = False):
        """
        Prints a warning message.
        """

        self._print_message(Logger.WARN, *info_message_arr, send_telegram=send_telegram)

    # endregion
